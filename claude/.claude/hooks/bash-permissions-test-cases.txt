# Regression tests for bash-permissions.json regex patterns.
#
# Deny/Paths layers — 3 columns:
#   EXPECT<tab>PATTERN_NAME<tab>COMMAND
#   EXPECT: DENY or PASS
#
# Allow layer — 4 columns:
#   EXPECT<tab>PATTERN_NAME<tab>BRANCH<tab>COMMAND
#   EXPECT: ALLOW or ASK
#   BRANCH: simulated branch name (- for branchless rules)
#
# Pattern name prefixes determine the layer:
#   path_*  → paths layer (sensitive file detection)
#   allow_* → allow layer (auto-accept on matching branch)
#   (other) → deny layer (tool preference enforcement)
#
# __HOME__ in commands is expanded to $HOME at runtime.
#
# Run with: bash bash-permissions-test.sh
#
# -------------------------------------------------------------------
# sed — (^|[|;&(])\s*(env\s+)?([A-Z_]+=\S*\s+)*sed\s+
# Denies sed in command position (with optional env/env-var prefixes).
# Passes when "sed" appears inside strings, arguments, or filenames.
# -------------------------------------------------------------------
# Should DENY: sed in command position
DENY	sed	sed -i 's/foo/bar/' file.txt
DENY	sed	sed -E 's/old/new/g' input.txt
DENY	sed	cat file | sed 's/x/y/'
DENY	sed	cat file |sed 's/x/y/'
DENY	sed	cd dir && sed -i 's/a/b/' f
DENY	sed	grep foo file; sed 's/a/b/' file
DENY	sed	$(sed 's/old/new/' file)
DENY	sed	result=$(sed -n '5p' file)
DENY	sed	(sed 's/foo/bar/' file)
DENY	sed	cat f | sed -n '/start/,/end/p'
DENY	sed	cat file | grep pattern | sed 's/old/new/'
# Should DENY: env-var prefixed sed
DENY	sed	LC_ALL=C sed -i '' 's/foo/bar/' file
DENY	sed	VAR=val sed -i 's/a/b/' file
DENY	sed	LANG=en_US.UTF-8 LC_ALL=C sed -i 's/x/y/' f
DENY	sed	cat f | LC_ALL=C sed 's/a/b/'
# Should DENY: env command prefix
DENY	sed	env VAR=1 sed -n '1p' file
DENY	sed	env LC_ALL=C sed -i 's/a/b/' file
DENY	sed	cat f | env LANG=C sed 's/x/y/'
# Should PASS: sed inside strings, arguments, filenames
PASS	sed	git commit -m "replaced sed with Edit tool"
PASS	sed	git commit -m "use sed for stream editing"
PASS	sed	git commit -m "refactor: replace sed -i with Edit tool for safer edits"
PASS	sed	npm test -- --grep "sed replacement"
PASS	sed	grep -r "sed" docs/
PASS	sed	ls docs/sed-reference.md
PASS	sed	cat CHANGELOG.md
PASS	sed	man sed
PASS	sed	git log --format="%H sed changes"
PASS	sed	npm install --save-dev @types/sed-like-pkg
#
# -------------------------------------------------------------------
# echo > — \becho\s+[^;&|]*>
# Denies echo with a direct redirect to file.
# Passes when the redirect belongs to a later command after a separator.
# -------------------------------------------------------------------
# Should DENY: echo with direct redirect
DENY	echo	echo "hello world" > file.txt
DENY	echo	echo "content" >> output.log
DENY	echo	echo 'data' > /tmp/test.txt
DENY	echo	echo -e "line1\nline2" > multi.txt
DENY	echo	echo "${VAR}" > config.txt
DENY	echo	echo foo>bar.txt
DENY	echo	for f in *.txt; do echo "header" > "$f"; done
# Known limitation: literal > inside quotes is indistinguishable from a
# redirect via regex. These are false positives we accept because shell
# quoting cannot be parsed reliably without a full parser.
DENY	echo	echo "a > b"
# Should PASS: no redirect, or redirect after a command separator
PASS	echo	echo "Building..." && npm test > output.log
PASS	echo	echo "status" ; command > file
PASS	echo	echo "status" | cat > file
PASS	echo	echo "hello world"
PASS	echo	echo "Running tests..."
PASS	echo	echo "step 1 done" && echo "step 2" && npm run build > build.log
PASS	echo	echo "done"; npm run lint > lint-results.txt
PASS	echo	echo "compiling" && gcc main.c -o main > compile.log
PASS	echo	node -e "console.log('echo test')"
#
# -------------------------------------------------------------------
# printf > — \bprintf\s+[^;&|]*>
# Denies printf with a direct redirect to file.
# Passes when the redirect belongs to a later command after a separator.
# -------------------------------------------------------------------
# Should DENY: printf with direct redirect
DENY	printf	printf '%s\n' "hello" > file.txt
DENY	printf	printf "content" >> output.log
DENY	printf	printf '%d items' 5 > count.txt
DENY	printf	$(printf '%s' "data" > /tmp/out)
# Should PASS: no redirect, or redirect after a command separator
PASS	printf	printf "Building..." && npm test > output.log
PASS	printf	printf "status" ; command > file
PASS	printf	printf "status" | cat > file
PASS	printf	printf '%s\n' "hello"
PASS	printf	printf "Running tests..."
#
# ===================================================================
# PATHS LAYER — sensitive file access detection (security critical)
# False negatives are security gaps.
# ===================================================================
#
# -------------------------------------------------------------------
# path_env_bare — (^|\s)\.env($|\s|[^a-zA-Z0-9])
# Catches bare .env references. The [^a-zA-Z0-9] alternate catches
# .env.production, .env.local, etc. (dot is non-alphanumeric).
# -------------------------------------------------------------------
# Should DENY: bare .env and .env.* variants
DENY	path_env_bare	cat .env
DENY	path_env_bare	source .env
DENY	path_env_bare	cp .env .env.bak
DENY	path_env_bare	ls -la .env
DENY	path_env_bare	cat .env.production
DENY	path_env_bare	rm .env.local
DENY	path_env_bare	docker run --env-file .env app
# Should PASS: similarly-named files that aren't .env
PASS	path_env_bare	cat .envrc
PASS	path_env_bare	ls .environment
PASS	path_env_bare	cat .env2
PASS	path_env_bare	source .envextra
#
# -------------------------------------------------------------------
# path_env_pathed — (^|\s)[^ ]*/\.env($|\s|[^a-zA-Z0-9])
# Catches pathed .env references (absolute, relative, nested).
# -------------------------------------------------------------------
# Should DENY: pathed .env references
DENY	path_env_pathed	cat /app/.env
DENY	path_env_pathed	cat ./config/.env
DENY	path_env_pathed	source /home/deploy/.env.production
DENY	path_env_pathed	cp ../project/.env .
DENY	path_env_pathed	ls /opt/app/.env.local
# Should PASS: pathed non-.env files
PASS	path_env_pathed	cat /app/.envrc
PASS	path_env_pathed	ls ./config/.environment
PASS	path_env_pathed	cat /tmp/.env2
#
# -------------------------------------------------------------------
# path_secrets — /secrets/
# Catches any reference to a /secrets/ directory.
# -------------------------------------------------------------------
# Should DENY: paths containing /secrets/
DENY	path_secrets	cat /app/secrets/db.yml
DENY	path_secrets	ls /secrets/api-key
DENY	path_secrets	cp config/secrets/token.json .
DENY	path_secrets	cat ./secrets/password.txt
# Should PASS: "secrets" without directory separators
PASS	path_secrets	cat secretsfile.txt
PASS	path_secrets	grep secrets README.md
PASS	path_secrets	ls /my-other-dir/
#
# -------------------------------------------------------------------
# path_cert_key — [^. ][^ ]*\.(pem|key)($|\s)
# Catches certificate and private key files. Requires at least one
# non-dot, non-space character before the extension to avoid matching
# bare .pem or .key (which aren't real filenames).
# -------------------------------------------------------------------
# Should DENY: certificate/key file references
DENY	path_cert_key	cat server.pem
DENY	path_cert_key	cat id_rsa.key
DENY	path_cert_key	openssl x509 -in cert.pem -text
DENY	path_cert_key	cp my.key /backup/
DENY	path_cert_key	ls *.pem
DENY	path_cert_key	scp primary.key user@host:/tmp/
# Should PASS: not cert/key files
PASS	path_cert_key	cat .pem
PASS	path_cert_key	cat .key
PASS	path_cert_key	grep key config.yml
PASS	path_cert_key	echo "primary key lookup"
PASS	path_cert_key	cat monkey.txt
PASS	path_cert_key	cat server.pemfile
#
# -------------------------------------------------------------------
# path_aws — (^|\s)(~|__HOME__)/\.aws/
# Catches AWS credential directory access via ~ or $HOME.
# -------------------------------------------------------------------
# Should DENY: home-relative .aws access
DENY	path_aws	cat ~/.aws/credentials
DENY	path_aws	cat ~/.aws/config
DENY	path_aws	ls ~/.aws/
DENY	path_aws	cat __HOME__/.aws/credentials
DENY	path_aws	ls __HOME__/.aws/config
# Should PASS: non-home .aws or unrelated commands
PASS	path_aws	cat .aws/credentials
PASS	path_aws	aws configure list
PASS	path_aws	echo "check aws settings"
#
# -------------------------------------------------------------------
# path_ssh — (^|\s)(~|__HOME__)/\.ssh/
# Catches SSH key/config directory access via ~ or $HOME.
# -------------------------------------------------------------------
# Should DENY: home-relative .ssh access
DENY	path_ssh	cat ~/.ssh/id_rsa
DENY	path_ssh	ssh-keygen -f ~/.ssh/id_ed25519
DENY	path_ssh	ls ~/.ssh/
DENY	path_ssh	cat __HOME__/.ssh/authorized_keys
DENY	path_ssh	chmod 600 ~/.ssh/config
# Should PASS: non-home .ssh or unrelated commands
PASS	path_ssh	cat .ssh/config
PASS	path_ssh	ssh user@host
PASS	path_ssh	echo "ssh key rotation"
#
# -------------------------------------------------------------------
# path_shell_config — (^|\s)(~|__HOME__)/\.(zshrc|bashrc|bash_profile|gitconfig)($|\s)
# Catches shell and git config file access in home directory.
# -------------------------------------------------------------------
# Should DENY: home shell/git config files
DENY	path_shell_config	cat ~/.zshrc
DENY	path_shell_config	vim ~/.bashrc
DENY	path_shell_config	source ~/.bash_profile
DENY	path_shell_config	cat ~/.gitconfig
DENY	path_shell_config	cat __HOME__/.zshrc
# Should PASS: non-home or suffixed variants
PASS	path_shell_config	cat ~/.zshrc.bak
PASS	path_shell_config	cat .zshrc
PASS	path_shell_config	git config --global user.name
PASS	path_shell_config	cat ~/.zshenv
#
# -------------------------------------------------------------------
# path_local_config — (^|\s)(~|__HOME__)/\.[a-z]+\.local($|\s)
# Catches private *.local override files in home directory.
# These contain machine-specific secrets/config that shouldn't be shared.
# -------------------------------------------------------------------
# Should DENY: home *.local dotfiles
DENY	path_local_config	cat ~/.zshrc.local
DENY	path_local_config	cat ~/.gitconfig.local
DENY	path_local_config	source ~/.bashrc.local
DENY	path_local_config	cat __HOME__/.npmrc.local
DENY	path_local_config	cat ~/.vimrc.local
# Should PASS: non-.local or non-home
PASS	path_local_config	cat ~/.zshrc
PASS	path_local_config	cat .zshrc.local
PASS	path_local_config	cat ~/something.local
PASS	path_local_config	cat ~/.Foo.local

# ===================================================================
# ALLOW LAYER — auto-accept trusted patterns (UX critical)
# ALLOW = hook exits 0 silently (auto-accepted).
# ASK = rule doesn't match; falls through to ask layer.
# ===================================================================
#
# -------------------------------------------------------------------
# allow_curl_local — localhost/127.0.0.1 curl requests
# No branch condition — always allowed.
# -------------------------------------------------------------------
# Should ALLOW: curl to localhost/loopback
ALLOW	allow_curl_local	-	curl http://localhost:3000/api/health
ALLOW	allow_curl_local	-	curl https://localhost:8080/
ALLOW	allow_curl_local	-	curl 'http://127.0.0.1:9090/status'
ALLOW	allow_curl_local	-	curl "http://localhost/path"
ALLOW	allow_curl_local	-	curl http://localhost:3000
ALLOW	allow_curl_local	-	curl -v http://localhost:3000/
ALLOW	allow_curl_local	-	rtk curl http://localhost:3000/api
# Should ASK: non-localhost curl or non-curl commands
ASK	allow_curl_local	-	curl https://example.com/api
ASK	allow_curl_local	-	curl http://production-server:3000/
ASK	allow_curl_local	-	curl http://localhostevil.com/
#
# -------------------------------------------------------------------
# allow_git_ops — git commit/push on personal feature branches
# Branch condition: ^mjb-pho-NEB-
# -------------------------------------------------------------------
# Should ALLOW: git commit/push on matching branch
ALLOW	allow_git_ops	mjb-pho-NEB-1234	git commit -m "fix: resolve issue"
ALLOW	allow_git_ops	mjb-pho-NEB-5678	git push
ALLOW	allow_git_ops	mjb-pho-NEB-9999	git push origin mjb-pho-NEB-9999
ALLOW	allow_git_ops	mjb-pho-NEB-1234	rtk git commit -m "fix"
# Should ASK: git commit/push on non-matching branch
ASK	allow_git_ops	main	git commit -m "fix"
ASK	allow_git_ops	develop	git push
ASK	allow_git_ops	feature/other	git commit -m "test"
# Should ASK: non-commit/push git commands on any branch
ASK	allow_git_ops	mjb-pho-NEB-1234	git stash drop
ASK	allow_git_ops	mjb-pho-NEB-1234	git rebase main
#
